function doPost(e) {
  var sheet = SpreadsheetApp.openById("11Sx4wr90wcLj-F4tq1JvY0lWse8W5nqe").getSheetByName("フォームの回答 1");
  var data = JSON.parse(e.postData.contents);
  sheet.appendRow([new Date(), data.groupCount, data.participantCount, data.asset, data.eventName]);
  return ContentService.createTextOutput("ok");
}

function doGet(e) {
  var sheet = SpreadsheetApp.openById("11Sx4wr90wcLj-F4tq1JvY0lWse8W5nqe").getSheetByName("フォームの回答 1");
  var eventName = e.parameter.event || "";
  var data = sheet.getDataRange().getValues();
  var totalGroups = 0;
  var totalPeople = 0;
  for (var i = 1; i < data.length; i++) {
    if (data[i][4] == eventName) {
      totalGroups = totalGroups + Number(data[i][1] || 0);
      totalPeople = totalPeople + Number(data[i][2] || 0);
    }
  }
  var result = JSON.stringify({groups: totalGroups, people: totalPeople});
  return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON);
}
