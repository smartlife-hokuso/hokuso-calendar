var FOLDER_ID = '1DlLDfaNYv5OHF96RO8qv2906wDQnJbA6';

function doPost(e) {
  var data = JSON.parse(e.postData.contents);
  var parentFolder = DriveApp.getFolderById(FOLDER_ID);

  var subFolderName = data.folderName;
  var subFolders = parentFolder.getFoldersByName(subFolderName);
  var folder;
  if (subFolders.hasNext()) {
    folder = subFolders.next();
  } else {
    folder = parentFolder.createFolder(subFolderName);
  }

  var blob = Utilities.newBlob(
    Utilities.base64Decode(data.base64),
    data.mimeType,
    data.fileName
  );
  var file = folder.createFile(blob);
  file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);

  return ContentService.createTextOutput(JSON.stringify({
    success: true,
    url: file.getUrl()
  })).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  var folderName = (e.parameter.folder || "").toString().trim();
  if (!folderName) {
    return ContentService.createTextOutput(JSON.stringify({found: false})).setMimeType(ContentService.MimeType.JSON);
  }

  var parentFolder = DriveApp.getFolderById(FOLDER_ID);

  // 完全一致で探す
  var subFolders = parentFolder.getFoldersByName(folderName);

  // 見つからなければ部分一致で探す
  if (!subFolders.hasNext()) {
    var allFolders = parentFolder.getFolders();
    while (allFolders.hasNext()) {
      var f = allFolders.next();
      var fName = f.getName();
      if (fName.indexOf(folderName) >= 0 || folderName.indexOf(fName) >= 0) {
        subFolders = parentFolder.getFoldersByName(fName);
        break;
      }
    }
  }

  if (!subFolders.hasNext()) {
    return ContentService.createTextOutput(JSON.stringify({found: false})).setMimeType(ContentService.MimeType.JSON);
  }

  var folder = subFolders.next();
  var files = folder.getFiles();
  var latestFile = null;
  var latestDate = null;

  while (files.hasNext()) {
    var file = files.next();
    var name = file.getName();
    // 元のpresentation.pdfとreport.html以外のファイルを探す
    if (name === "presentation.pdf" || name === "report.html") continue;
    var created = file.getDateCreated();
    if (!latestDate || created > latestDate) {
      latestDate = created;
      latestFile = file;
    }
  }

  if (latestFile) {
    var d = latestFile.getDateCreated();
    var dateStr = Utilities.formatDate(d, "Asia/Tokyo", "yyyy/MM/dd");
    return ContentService.createTextOutput(JSON.stringify({
      found: true,
      url: latestFile.getUrl(),
      name: latestFile.getName(),
      date: dateStr
    })).setMimeType(ContentService.MimeType.JSON);
  }

  return ContentService.createTextOutput(JSON.stringify({found: false})).setMimeType(ContentService.MimeType.JSON);
}
